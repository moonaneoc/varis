Type judgment for javascript

### Usage

```js
    const varis = require("varis");

    varis.isStr("val");
    varis.isNum("val");
    varis.isInt("val");
    varis.isArr("val");
    varis.isFunc("val");
    varis.isBool("val");
    varis.isNull("val");
    varis.isUndef("val");
    varis.isDate("val"); // instanceof Date
    varis.isObj("val");
    varis.isFormatDate("2019-03-22 10:21:46.323"); // "2019-03-22 10:21:46","2019-03-22 10:21:46.323"
```
